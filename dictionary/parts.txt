## Get Inventory Part ##
select 
    objid, 
    PART_NO, 
    &AO.Inventory_Part_API.Get_Description(CONTRACT, PART_NO), 
    CONTRACT, 
    DESCRIPTION_COPY, 
    &AO.Part_Catalog_API.Get_Description(PART_NO)      
from 
    &AO.INVENTORY_PART 
where 
    (PART_NO = :p0)    
    


## Get Purchase Part ##
select 
    objid, 
    PART_NO, 
    &AO.Purchase_Part_API.Get_Description(contract, part_no), 
    CONTRACT
from 
    &AO.PURCHASE_PART 
where 
    (PART_NO = :p0)            



## Get Sales Part ##       
select 
    objid, 
    CATALOG_NO, 
    &AO.Sales_Part_API.Get_Catalog_Desc(CONTRACT, CATALOG_NO), 
    CONTRACT
from 
    &AO.SALES_PART 
where 
    CATALOG_TYPE_DB = 'INV' 
and 
    (CATALOG_NO = :p0)

    AQUISITION_CODE, 
    PLANNING_METHOD, 
    CONFIGURABLE, 
    LOT_TRACKING_CODE, 
    SERIAL_TYPE, 
    SERIAL_TRACKING_CODE,

from 
    &AO.ENG_PART_MASTER_MAIN

    (PART_NO = :p0)

    &AO.INVENTORY_PART where (PART_NO = :p0)
